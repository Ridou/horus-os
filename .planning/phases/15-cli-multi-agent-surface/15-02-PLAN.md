## Phase 14 streaming API (verified in src/horus_os/agent.py)

```
async def run_agent_stream(
    prompt: str,
    *,
    provider: str = "anthropic",
    model: str | None = None,
    max_tokens: int = 1024,
    system: str | None = None,
) -> AsyncGenerator[str | ToolCallEvent, None]:
```

The generator yields `str` tokens incrementally, then optionally a
`ToolCallEvent` after the text stream completes. The CLI consumer only needs
to print `str` items live and ignore (or surface as a separate line) any
ToolCallEvent objects. Tool dispatch is intentionally not handled here.

## asyncio.run from a sync CLI handler (research §Pitfall 1, §Pattern 3)

`asyncio.run()` is the correct bridge. The sync handler stays sync; only the
streaming branch wraps the consumer coroutine in `asyncio.run(...)`. CLI tests
are also sync, so no nested event loop hazard.

## Precedence rules (research §Pitfall 4)

`model = args.model or (profile.default_model if profile else None) or _model_for(config, provider)`

User-supplied `--model` always wins over profile default. Profile default wins
over config default. This is the same fallback chain `_model_for` already
implements, with the profile slot inserted in the middle.

## Trace recording on the streaming path (research §Pitfall 3)

The buffered path calls `db.record_trace(prompt, result, latency_ms=...)`. The
streaming path must do the equivalent: assemble the streamed text into an
AgentResult and call record_trace with the same latency, status, and
error_message semantics. The agent_profile_name field (Phase 13 v4 schema) is
populated when --agent was used.

## Test pattern for async generator fakes (research §Pattern 4)

```
async def fake_stream(prompt, *, provider, model, max_tokens=1024, system=None):
    for token in ["Hello", ", ", "world"]:
        yield token
```

Tests monkeypatch `run_cmd.run_agent_stream` to swap in `fake_stream`. The
sync test calls `main(argv, stdout=..., stderr=...)` and asserts stdout
contains the joined tokens. `io.StringIO.flush()` is a no-op and does not
raise.

def _run_buffered(
    prompt,
    *,
    provider,
    model,
    max_iterations,
    record,
    system_prompt,
    agent_name,
    registry,
    db,
    stdout,
    stderr,
) -> int:
    tool_log: list[ToolResult] = []
    start = time.perf_counter()
    try:
        result = run_agent_loop(
            prompt,
            registry=registry,
            provider=provider,
            model=model,
            max_iterations=max_iterations,
            system_prompt=system_prompt,
            on_tool_result=tool_log.append,
        )
        latency_ms = int((time.perf_counter() - start) * 1000)
        if record:
            db.record_trace(prompt, result, latency_ms=latency_ms, agent_profile_name=agent_name)
        _print_result(stdout, result, latency_ms, tool_log)
        return 0
    except Exception as exc:
        latency_ms = int((time.perf_counter() - start) * 1000)
        if record:
            db.record_trace(
                prompt,
                AgentResult(text="", tool_uses=[], provider=provider, model=model, usage={}),
                latency_ms=latency_ms,
                status="error",
                error_message=f"{type(exc).__name__}: {exc}",
                agent_profile_name=agent_name,
            )
        stderr.write(f"Agent run failed: {type(exc).__name__}: {exc}\n")
        return 1
```

def _run_streaming(
    prompt,
    *,
    provider,
    model,
    record,
    system_prompt,
    agent_name,
    db,
    stdout,
    stderr,
) -> int:
    text_parts: list[str] = []
    start = time.perf_counter()
    try:
        asyncio.run(
            _consume_stream(
                prompt,
                provider=provider,
                model=model,
                system_prompt=system_prompt,
                text_parts=text_parts,
                stdout=stdout,
                stderr=stderr,
            )
        )
        latency_ms = int((time.perf_counter() - start) * 1000)
        stdout.write(f"\n\n[{provider}/{model}, {latency_ms}ms, streamed]\n")
        if record:
            db.record_trace(
                prompt,
                AgentResult(
                    text="".join(text_parts),
                    tool_uses=[],
                    provider=provider,
                    model=model,
                    usage={},
                ),
                latency_ms=latency_ms,
                agent_profile_name=agent_name,
            )
        return 0
    except Exception as exc:
        latency_ms = int((time.perf_counter() - start) * 1000)
        if record:
            db.record_trace(
                prompt,
                AgentResult(text="".join(text_parts), tool_uses=[], provider=provider, model=model, usage={}),
                latency_ms=latency_ms,
                status="error",
                error_message=f"{type(exc).__name__}: {exc}",
                agent_profile_name=agent_name,
            )
        stderr.write(f"Agent run failed: {type(exc).__name__}: {exc}\n")
        return 1


async def _consume_stream(
    prompt,
    *,
    provider,
    model,
    system_prompt,
    text_parts,
    stdout,
    stderr,
) -> None:
    async for chunk in run_agent_stream(
        prompt,
        provider=provider,
        model=model,
        system=system_prompt,
    ):
        if isinstance(chunk, ToolCallEvent):
            stderr.write(f"[tool-request] {chunk.name}({chunk.input})\n")
            continue
        text_parts.append(chunk)
        stdout.write(chunk)
        stdout.flush()
```

Note: ToolCallEvent emission is informational; the streaming path does not
dispatch tools by design (matching the Phase 14 contract). The CLI surfaces
the event to stderr so it does not pollute the response body on stdout.
