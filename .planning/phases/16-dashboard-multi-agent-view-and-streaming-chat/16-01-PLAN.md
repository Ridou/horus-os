## Last-activity query

A scalar SQL: `SELECT MAX(created_at) FROM traces WHERE agent_profile_name = ?`.
Add a `_last_activity_for` helper inside `api.py` rather than a method
on `Database`, since this is the only consumer. Returns ISO string or
None.

## Agents JSON payload shape

```
{
  "name": "default",
  "system_prompt": "...",
  "default_model": null,
  "allowed_tools": null,
  "memory_scope": null,
  "created_at": "...",
  "updated_at": "...",
  "last_activity_at": "..." | null
}
```

`allowed_tools=None` means unrestricted (matches CLI semantics). The
list view returns `{"agents": [<payload>, ...]}`. The show view returns
the payload bare. `last_activity_at` only appears on list and show
responses; it is derived, not stored.

## Edit semantics

PATCH accepts a subset of fields. Any field not in the JSON body keeps
its current value. `allowed_tools` accepts the same three modes as the
CLI: a list (explicit), `null` (clear restrictions), or absent
(unchanged). To distinguish "absent" from "null" in JSON, use
`payload.get(field)` and a sentinel check on the raw `payload` dict.

## Backwards compat hard rule

Every endpoint that returns trace data must run through `_trace_to_dict`.
That function already calls `dataclasses.asdict(record)`, which
includes the new fields with their dataclass defaults (None). v0.1
trace rows where the SQLite columns are literally NULL flow through
`Database._row_to_trace` and end up with `parent_trace_id=None`,
`agent_profile_name=None`. No frontend assumptions are needed.

## SSE error frame contract

When the consumer crashes inside the async generator, we cannot raise
an HTTPException (the response is already in flight). The pattern is:

```
try:
    async for chunk in run_agent_stream(...):
        yield ...
except Exception as exc:
    yield f"data: {json.dumps({'type':'error','message':...})}\n\n"
    record an error trace synchronously
    return
finally:
    if not errored: record a success trace, yield done frame
```

We thread an `errored: list[Exception] = []` cell through the generator
because async generators do not have a clean post-iteration hook in
FastAPI's StreamingResponse model.

def _profile_to_dict(profile: AgentProfile, *, last_activity_at: str | None) -> dict[str, Any]:
    return {
        "name": profile.name,
        "system_prompt": profile.system_prompt,
        "default_model": profile.default_model,
        "allowed_tools": list(profile.allowed_tools) if profile.allowed_tools is not None else None,
        "memory_scope": profile.memory_scope,
        "created_at": profile.created_at,
        "updated_at": profile.updated_at,
        "last_activity_at": last_activity_at,
    }


def _last_activity_for(db: Database, name: str) -> str | None:
    with db._connect() as conn:
        row = conn.execute(
            "SELECT MAX(created_at) AS ts FROM traces WHERE agent_profile_name = ?",
            (name,),
        ).fetchone()
        return row["ts"] if row is not None else None
```

helper. Using it directly is consistent with how other helpers
(e.g., the chat handler's `_persist_write` callback) reach into
the database. The cursor is read-only, no schema risk.

@app.post("/api/chat/stream")
async def chat_stream(payload: dict = Body(...)) -> StreamingResponse:
    prompt = (payload.get("prompt") or "").strip() if isinstance(payload, dict) else ""
    if not prompt:
        raise HTTPException(400, detail="prompt is required")
    cfg = _config()
    if not cfg.db_path.exists():
        raise HTTPException(503, detail="database not initialized; run `horus-os init`")
    provider = payload.get("provider") or cfg.default_provider
    if provider not in SUPPORTED_PROVIDERS:
        raise HTTPException(400, detail=f"unknown provider {provider!r}")
    if not _api_key_for(provider):
        env_var = "ANTHROPIC_API_KEY" if provider == "anthropic" else "GEMINI_API_KEY"
        raise HTTPException(503, detail=f"no API key for {provider}; set {env_var}")
    db = Database(cfg.db_path)
    agent_name = payload.get("agent")
    profile: AgentProfile | None = None
    if isinstance(agent_name, str) and agent_name:
        profile = db.load_profile(agent_name)
        if profile is None:
            raise HTTPException(404, detail=f"agent profile {agent_name!r} not found")
    else:
        agent_name = None
    model = (
        payload.get("model")
        or (profile.default_model if profile else None)
        or _model_for(cfg, provider)
    )
    system_prompt = profile.system_prompt if profile else None

    async def _event_stream() -> AsyncGenerator[bytes, None]:
        from horus_os.agent import run_agent_stream as _stream

        text_parts: list[str] = []
        start = time.perf_counter()
        try:
            async for chunk in _stream(
                prompt,
                provider=provider,
                model=model,
                system=system_prompt,
            ):
                if isinstance(chunk, ToolCallEvent):
                    yield _sse({"type": "tool_call", "name": chunk.name, "input": chunk.input})
                    continue
                text_parts.append(chunk)
                yield _sse({"type": "token", "text": chunk})
        except Exception as exc:
            latency_ms = int((time.perf_counter() - start) * 1000)
            trace_id = db.record_trace(
                prompt,
                AgentResult(
                    text="".join(text_parts),
                    tool_uses=[],
                    provider=provider,
                    model=model,
                    usage={},
                ),
                latency_ms=latency_ms,
                status="error",
                error_message=f"{type(exc).__name__}: {exc}",
                agent_profile_name=agent_name,
            )
            yield _sse(
                {
                    "type": "error",
                    "message": f"{type(exc).__name__}: {exc}",
                    "trace_id": trace_id,
                }
            )
            return
        latency_ms = int((time.perf_counter() - start) * 1000)
        trace_id = db.record_trace(
            prompt,
            AgentResult(
                text="".join(text_parts),
                tool_uses=[],
                provider=provider,
                model=model,
                usage={},
            ),
            latency_ms=latency_ms,
            agent_profile_name=agent_name,
        )
        yield _sse({"type": "done", "trace_id": trace_id, "latency_ms": latency_ms})

    return StreamingResponse(_event_stream(), media_type="text/event-stream")
```

    return TestClient(create_app(data_dir=tmp_path))


def _frames(response_bytes: bytes) -> list[dict[str, Any]]:
    """Split SSE response body into a list of decoded JSON payloads."""
    out: list[dict[str, Any]] = []
    for raw in response_bytes.split(b"\n\n"):
        line = raw.strip()
        if not line.startswith(b"data: "):
            continue
        out.append(json.loads(line[len(b"data: "):].decode("utf-8")))
    return out
